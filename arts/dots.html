import nimsvg, 
       os,
       random
     
let settings = animSettings("./" / sourceBaseName(), backAndForth=true)
let numFrames = 100
     
settings.buildAnimation(numFrames) do (i: int) -> Nodes:
    let size = 500
    buildSvg:
        svg(width=size, height=size):
            for _ in 0 .. 500:
                let x = random(size)
                let y = random(size)
                let radius = random(5)
                circle(cx=x, cy=y, r=radius, stroke="#00e353", fill="#0b4a22", `fill-opacity`=0.5)
