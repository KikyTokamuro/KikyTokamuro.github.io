package main

import (
    "math"
    "math/rand"
            
    "github.com/fogleman/gg"
)
            
func main() {
    const S = 1024
    const N = 2048
    dc := gg.NewContext(S, S)

    dc.SetRGB(1, 1, 1)
    dc.Clear()
            
    for i := 0; i <= N; i++ {
        t := float64(i) / N * 5
        d := t*S*0.4 + 25
        a := t * math.Pi * 4 * 100
        x := S/2 + math.Cos(a)*d
        y := S/2 + math.Sin(a)*d    
        for i := 1; i < 20; i += 5 {
            r := rand.Float64()
            g := rand.Float64()
            b := rand.Float64()
            aa := rand.Float64()
            dc.SetRGBA(r, g, b, aa) 
            dc.DrawCircle(x+float64(i), y, 6)
            dc.Fill()
        }
    }
            
    dc.SavePNG("spiraldots.png")
}
